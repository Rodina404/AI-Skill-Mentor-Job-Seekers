// ─── Extract raw text from the uploaded file ─────────────────
// PDF → PDF.js (runs in browser, no server needed)
// TXT → FileReader
// DOCX → tell user to convert
async function extractTextFromFile(file) {
  const name = file.name.toLowerCase();

  if (name.endsWith('.txt')) {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload  = e  => resolve(e.target.result);
      reader.onerror = () => reject(new Error('Could not read TXT file'));
      reader.readAsText(file);
    });
  }

  if (name.endsWith('.pdf')) {
    // Use PDF.js to render every page and collect its text content
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onerror = () => reject(new Error('Could not read PDF file'));
      reader.onload = async (e) => {
        try {
          const typedArray = new Uint8Array(e.target.result);
          const pdf = await pdfjsLib.getDocument({ data: typedArray }).promise;
          let fullText = '';
          for (let p = 1; p <= pdf.numPages; p++) {
            const page    = await pdf.getPage(p);
            const content = await page.getTextContent();
            const pageText = content.items.map(item => item.str).join(' ');
            fullText += pageText + '\n';
          }
          if (!fullText.trim()) reject(new Error('PDF has no extractable text (may be image-only scan)'));
          else resolve(fullText);
        } catch(err) { reject(err); }
      };
      reader.readAsArrayBuffer(file);
    });
  }

  if (name.endsWith('.docx') || name.endsWith('.doc')) {
    throw new Error('DOCX/DOC cannot be read in the browser. Please save your resume as PDF or TXT and re-upload.');
  }

  throw new Error('Unsupported file type: ' + file.name);
}

const SYSTEM_PROMPT = `You are a resume parser. Extract structured data from the resume text below and return ONLY a JSON object — no markdown, no explanation, nothing else.

Required JSON fields:
{
  "name": "exact full name as written in the resume",
  "title": "most recent or target job title",
  "email": "email address or empty string",
  "phone": "phone number or empty string",
  "education": ["one string per degree: Degree at Institution (Year)"],
  "experience": ["one string per role: Job Title at Company (Year-Year or present)"],
  "skills": ["every skill, tool, language, framework, certification found anywhere in the document"],
  "experienceYears": <integer: total years of work experience>,
  "experienceScore": <float 0-1: 0yr=0.2, 1yr=0.3, 1-3yr=0.4, 3-5yr=0.6, 5-8yr=0.75, 8+yr=0.9>,
  "educationScore": <float 0-1: high_school=0.3, diploma=0.4, bachelor=0.6, master=0.8, phd=1.0>
}

Critical rules:
- name MUST be the real person's name from the document
- Extract every single skill — be thorough, include tools, languages, soft skills, certifications
- Return raw JSON only, no code fences`;

async function callClaude(resumeText) {
  const resp = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 1500,
      system: SYSTEM_PROMPT,
      messages: [{
        role: 'user',
        content: 'Parse this resume:\n\n' + resumeText.slice(0, 9000)
      }]
    })
  });

  if (!resp.ok) {
    let msg = `API error ${resp.status}`;
    try { const b = await resp.json(); if (b?.error?.message) msg = b.error.message; } catch {}
    throw new Error(msg);
  }

  const data = await resp.json();
  const raw  = (data.content?.find(b => b.type === 'text')?.text || '').trim();
  const clean = raw.replace(/^```json\s*/,'').replace(/^```\s*/,'').replace(/```\s*$/,'').trim();

  let parsed;
  try { parsed = JSON.parse(clean); }
  catch { throw new Error('Could not parse AI response — please retry'); }

  // Sanitise fields
  if (!Array.isArray(parsed.skills))     parsed.skills     = [];
  if (!Array.isArray(parsed.education))  parsed.education  = [];
  if (!Array.isArray(parsed.experience)) parsed.experience = [];
  if (!parsed.name || parsed.name.trim().length < 2) {
    parsed.name = uploadedFile.name.replace(/\.[^.]+$/, '').replace(/[-_]/g, ' ')
      .replace(/resume|cv|curriculum|vitae/gi, '').trim();
  }

  return parsed;
}

function scanSkillsFromText(text) {

function guessProfileFromText(text, filename) {
  const lines = text.split(/\n/).map(l => l.trim()).filter(Boolean);

  // Name: first line that looks like "FirstName LastName"
  let name = '';
  for (const l of lines.slice(0, 10)) {
    if (/^[A-Z][a-z]+([ '-][A-Z][a-z]+)+$/.test(l) && l.split(/\s+/).length <= 4) {
      name = l; break;
    }
  }
  if (!name) name = filename.replace(/\.[^.]+$/, '').replace(/[-_]/g, ' ')
    .replace(/resume|cv|curriculum|vitae/gi, '').trim();

  // Email
  const emailM = text.match(/[\w.+'-]+@[\w-]+\.[a-z]{2,}/i);

  // Title: line with common job title keywords
  const titleM = lines.find(l => /analyst|engineer|scientist|developer|manager|designer|consultant|intern/i.test(l) && l.length < 60);

  // Education lines
  const eduKw = ['bachelor','master','phd','b.sc','m.sc','b.eng','mba','diploma','degree','university','college'];
  const edu = lines.filter(l => eduKw.some(k => l.toLowerCase().includes(k))).slice(0, 3);

  // Experience years
  const expM = text.match(/(\d+)\+?\s*years?\s*(?:of\s+)?(?:experience|exp|work)/i);
  const expYears = expM ? parseInt(expM[1]) : 0;
  const expScore = expYears >= 8 ? 0.9 : expYears >= 5 ? 0.75 : expYears >= 3 ? 0.6 : expYears >= 1 ? 0.4 : 0.3;

  const hasMaster  = /master|m\.sc|mba|phd|doctorate/i.test(text);
  const hasBachelor = /bachelor|b\.sc|b\.eng|undergraduate|bsc/i.test(text);
  const eduScore = hasMaster ? 0.8 : hasBachelor ? 0.6 : 0.4;

  return {
    name,
    title: titleM || '',
    email: emailM ? emailM[0] : '',
    phone: '',
    education: edu,
    experience: [],
    skills: scanSkillsFromText(text),
    experienceYears: expYears,
    experienceScore: expScore,
    educationScore: eduScore,
    _source: 'local keyword scan',
  };
}

function runPipeline() {
  const btn  = document.getElementById('run-btn');
  const spin = document.getElementById('run-spin');
  const lbl  = document.getElementById('run-label');
  btn.disabled = true;
  spin.style.display = 'block';
  lbl.textContent = 'Generating…';
  setStep(4);

  setTimeout(() => {
    try {
      const role         = document.getElementById('inp-role').value;
      const hoursPerWeek = +document.getElementById('inp-hours').value;
      const deadlineWeeks= +document.getElementById('inp-weeks').value;
      const expScore     = extractedProfile?.experienceScore || 0.4;
      const eduScore     = extractedProfile?.educationScore  || 0.6;
      const name         = extractedProfile?.name || 'You';

      // Gap detection: skill not covered = no current skill is substring or partial match
      const allRequired = ROLE_SKILLS[role] || [];
      const gaps = allRequired.filter(g => {
        const req = g.skill.toLowerCase();
        return !currentSkills.some(s => {
          const us = s.toLowerCase();
          return us === req || us.includes(req) || req.includes(us) || levenSim(us, req) > 0.7;
        });
      }).map(g => {
        // Partial match reduces gap score
        const bestSim = Math.max(0, ...currentSkills.map(s => levenSim(s.toLowerCase(), g.skill.toLowerCase())));
        if (bestSim > 0.4) {
          return { ...g, gap: +(g.gap * (1 - bestSim * 0.5)).toFixed(2), sim: +Math.min(0.8, bestSim).toFixed(2) };
        }
        return g;
      });

      const roadmap = buildRoadmap(gaps, hoursPerWeek, deadlineWeeks, name);

      renderGaps(gaps);
      renderRoadmap(roadmap);
      renderCourses(gaps);
      renderProgress(roadmap, gaps, expScore, eduScore, hoursPerWeek);
      renderNotifications(roadmap, gaps, name);
      renderExplainability(gaps, name);

      const badge = document.getElementById('output-badge');
      badge.textContent = gaps.length + ' gaps found';
      badge.className = 'badge badge-teal';
      setStep(5);
      switchTab('gaps');
    } catch(e) { console.error(e); }

    btn.disabled = false;
    spin.style.display = 'none';
    lbl.textContent = '🔄 Regenerate';
  }, 600);
}

function levenSim(a, b) {
  if (a === b) return 1;
  const longer = a.length > b.length ? a : b;
  const shorter = a.length > b.length ? b : a;
  if (!longer.length) return 1;
  const m = longer.length, n = shorter.length;
  const dp = Array.from({length:m+1},(_,i)=>Array.from({length:n+1},(_,j)=>i===0?j:j===0?i:0));
  for(let i=1;i<=m;i++) for(let j=1;j<=n;j++) dp[i][j]=longer[i-1]===shorter[j-1]?dp[i-1][j-1]:1+Math.min(dp[i-1][j],dp[i][j-1],dp[i-1][j-1]);
  return (longer.length - dp[m][n]) / longer.length;
}

function buildRoadmap(gaps, hoursPerWeek, deadlineWeeks, name) {
  const sorted = [...gaps].sort((a,b) => {
    const pw={high:3,medium:2,low:1};
    return (pw[b.priority]||2)*(1-b.sim) - (pw[a.priority]||2)*(1-a.sim);
  });
  const BUFFER_EVERY = 4;
  const remaining = {};
  sorted.forEach(g => { const c = COURSES[g.skill]; remaining[g.skill] = c ? c.hours : 20; });
  const weeks = [];
  let weekNum = 1;
  while (weekNum <= deadlineWeeks && Object.values(remaining).some(h => h > 0)) {
    const isBuffer = (weekNum % (BUFFER_EVERY + 1) === 0);
    let hoursLeft = isBuffer ? hoursPerWeek * 0.5 : hoursPerWeek;
    const tasks = [], skillsThisWeek = [];
    if (isBuffer) {
      tasks.push({type:'review', title:'Review & consolidation', hours:+(hoursPerWeek*0.4).toFixed(1)});
      tasks.push({type:'project',title:'End-of-module mini project',hours:+(hoursPerWeek*0.6).toFixed(1)});
    } else {
      for (const g of sorted) {
        if ((remaining[g.skill]||0) <= 0 || hoursLeft <= 0) continue;
        const chunk = +Math.min(remaining[g.skill], hoursLeft, hoursPerWeek*0.6).toFixed(1);
        if (chunk < 0.5) continue;
        const c = COURSES[g.skill] || {title:g.skill+' Course',platform:'Udemy',rating:4.5};
        tasks.push({type:'course',title:c.title.slice(0,46),skill:g.skill,hours:chunk,course:c});
        remaining[g.skill] = +(remaining[g.skill]-chunk).toFixed(2);
        hoursLeft = +(hoursLeft-chunk).toFixed(2);
        skillsThisWeek.push(g.skill);
        if (remaining[g.skill] <= 0) {
          const projH = +Math.min(hoursLeft,2).toFixed(1);
          if (projH >= 0.5) { tasks.push({type:'project',title:'Mini project: Apply '+g.skill,hours:projH,skill:g.skill}); hoursLeft=+(hoursLeft-projH).toFixed(2); }
        }
      }
    }
    const coveredCount = Object.values(remaining).filter(h=>h<=0).length;
    const pct = sorted.length ? coveredCount/sorted.length : 0;
    let milestone = null;
    if (Math.abs(pct-.25)<.12 && coveredCount>0) milestone='🏅 25% complete';
    else if (Math.abs(pct-.5)<.12) milestone='🏆 Halfway there!';
    else if (Math.abs(pct-.75)<.12) milestone='🚀 75% complete';
    else if (pct>=.99) milestone='🎓 All '+sorted.length+' skills done!';
    weeks.push({weekNum,isBuffer,tasks,milestone,theme:skillsThisWeek.slice(0,2).join(' + ')||'Review',totalHours:+tasks.reduce((s,t)=>s+t.hours,0).toFixed(1)});
    weekNum++;
  }
  return {name, weeks, totalWeeks:weeks.length, hoursPerWeek, deadlineWeeks, gaps:sorted};
}

function renderRoadmap(roadmap) {

// PDF → PDF.js (runs in browser, no server needed)
// TXT → FileReader
// DOCX → tell user to convert
async function extractTextFromFile(file) {
  const name = file.name.toLowerCase();

  if (name.endsWith('.txt')) {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload  = e  => resolve(e.target.result);
      reader.onerror = () => reject(new Error('Could not read TXT file'));
      reader.readAsText(file);
    });
  }

  if (name.endsWith('.pdf')) {
    // Use PDF.js to render every page and collect its text content
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onerror = () => reject(new Error('Could not read PDF file'));
      reader.onload = async (e) => {
        try {
          const typedArray = new Uint8Array(e.target.result);
          const pdf = await pdfjsLib.getDocument({ data: typedArray }).promise;
          let fullText = '';
          for (let p = 1; p <= pdf.numPages; p++) {
            const page    = await pdf.getPage(p);
            const content = await page.getTextContent();
            const pageText = content.items.map(item => item.str).join(' ');
            fullText += pageText + '\n';
          }
          if (!fullText.trim()) reject(new Error('PDF has no extractable text (may be image-only scan)'));
          else resolve(fullText);
        } catch(err) { reject(err); }
      };
      reader.readAsArrayBuffer(file);
    });
  }

  if (name.endsWith('.docx') || name.endsWith('.doc')) {
    throw new Error('DOCX/DOC cannot be read in the browser. Please save your resume as PDF or TXT and re-upload.');
  }

  throw new Error('Unsupported file type: ' + file.name);
}

const SYSTEM_PROMPT = `You are a resume parser. Extract structured data from the resume text below and return ONLY a JSON object — no markdown, no explanation, nothing else.

Required JSON fields:

- Extract every single skill — be thorough, include tools, languages, soft skills, certifications
- Return raw JSON only, no code fences`;

async function callClaude(resumeText) {
  const resp = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 1500,
      system: SYSTEM_PROMPT,
      messages: [{
        role: 'user',
        content: 'Parse this resume:\n\n' + resumeText.slice(0, 9000)
      }]
    })
  });

  if (!resp.ok) {
    let msg = `API error ${resp.status}`;
    try { const b = await resp.json(); if (b?.error?.message) msg = b.error.message; } catch {}
    throw new Error(msg);
  }

  const data = await resp.json();
  const raw  = (data.content?.find(b => b.type === 'text')?.text || '').trim();
  const clean = raw.replace(/^```json\s*/,'').replace(/^```\s*/,'').replace(/```\s*$/,'').trim();

  let parsed;
  try { parsed = JSON.parse(clean); }
  catch { throw new Error('Could not parse AI response — please retry'); }

  // Sanitise fields
  if (!Array.isArray(parsed.skills))     parsed.skills     = [];
  if (!Array.isArray(parsed.education))  parsed.education  = [];
  if (!Array.isArray(parsed.experience)) parsed.experience = [];
  if (!parsed.name || parsed.name.trim().length < 2) {
    parsed.name = uploadedFile.name.replace(/\.[^.]+$/, '').replace(/[-_]/g, ' ')
      .replace(/resume|cv|curriculum|vitae/gi, '').trim();
  }

  return parsed;
}

// Client-side keyword scan — used when API is unavailable
function scanSkillsFromText(text) {
  const tl = text.toLowerCase();
  return ALL_SKILLS.filter(s => tl.includes(s.toLowerCase()));
}

function guessProfileFromText(text, filename) {
  const lines = text.split(/\n/).map(l => l.trim()).filter(Boolean);

  // Name: first line that looks like "FirstName LastName"
  let name = '';
  for (const l of lines.slice(0, 10)) {
    if (/^[A-Z][a-z]+([ '-][A-Z][a-z]+)+$/.test(l) && l.split(/\s+/).length <= 4) {
      name = l; break;
    }
  }
  if (!name) name = filename.replace(/\.[^.]+$/, '').replace(/[-_]/g, ' ')
    .replace(/resume|cv|curriculum|vitae/gi, '').trim();

  // Email
  const emailM = text.match(/[\w.+'-]+@[\w-]+\.[a-z]{2,}/i);

  // Title: line with common job title keywords
  const titleM = lines.find(l => /analyst|engineer|scientist|developer|manager|designer|consultant|intern/i.test(l) && l.length < 60);

  // Education lines
  const eduKw = ['bachelor','master','phd','b.sc','m.sc','b.eng','mba','diploma','degree','university','college'];
  const edu = lines.filter(l => eduKw.some(k => l.toLowerCase().includes(k))).slice(0, 3);

  // Experience years
  const expM = text.match(/(\d+)\+?\s*years?\s*(?:of\s+)?(?:experience|exp|work)/i);
  const expYears = expM ? parseInt(expM[1]) : 0;
  const expScore = expYears >= 8 ? 0.9 : expYears >= 5 ? 0.75 : expYears >= 3 ? 0.6 : expYears >= 1 ? 0.4 : 0.3;

  const hasMaster  = /master|m\.sc|mba|phd|doctorate/i.test(text);
  const hasBachelor = /bachelor|b\.sc|b\.eng|undergraduate|bsc/i.test(text);
  const eduScore = hasMaster ? 0.8 : hasBachelor ? 0.6 : 0.4;

  return {
    name,
    title: titleM || '',
    email: emailM ? emailM[0] : '',
    phone: '',
    education: edu,
    experience: [],
    skills: scanSkillsFromText(text),
    experienceYears: expYears,
    experienceScore: expScore,
    educationScore: eduScore,
    _source: 'local keyword scan',
  };
}

// ─── MAIN ANALYZE FLOW ───────────────────────────────────────
async function analyzeResume() {
  if (!uploadedFile) {
    showStream('Please upload your resume first.', 'error');
    return;
  }

  const btn  = document.getElementById('analyze-btn');
  const spin = document.getElementById('analyze-spin');
  const lbl  = document.getElementById('analyze-label');
  btn.disabled = true;
  spin.style.display = 'block';
  lbl.textContent = 'Reading file…';
  setStep(2);

  const stream = document.getElementById('ai-stream');
  stream.classList.add('show');
  stream.innerHTML = '';

  function log(msg, type) {
    const color = type === 'error' ? 'var(--red-400)' : type === 'warn' ? 'var(--amber-600)' : 'var(--teal-600)';
    stream.innerHTML += `<div style="color:${color};margin:2px 0">${msg}</div>`;
    stream.scrollTop = stream.scrollHeight;
  }

  try {
    // ── Step A: Extract text from file (client-side, no server needed) ──
    log('→ Reading ' + uploadedFile.name + '…');
    lbl.textContent = 'Extracting text…';

    let resumeText;
    try {
      resumeText = await extractTextFromFile(uploadedFile);
      extractedText = resumeText;
      log('✓ Extracted ' + resumeText.length.toLocaleString() + ' characters from document');
    } catch (readErr) {
      log('✗ ' + readErr.message, 'error');
      throw readErr;
    }

    // ── Step B: Send text to Claude API ──
    log('→ Sending to Claude AI for analysis…');
    lbl.textContent = 'AI analyzing…';

    let profile;
    try {
      profile = await callClaude(resumeText);
      log('✓ AI extracted ' + profile.skills.length + ' skills, ' + profile.education.length + ' education entries');
    } catch (apiErr) {
      // API failed — fall back to local keyword scan so user still gets a result
      log('⚠ AI API unavailable (' + apiErr.message + ') — using local skill scanner', 'warn');
      profile = guessProfileFromText(resumeText, uploadedFile.name);
      log('✓ Local scan found ' + profile.skills.length + ' skills');
    }

    // ── Step C: Show profile ──
    setTimeout(() => { stream.classList.remove('show'); }, 1800);

    extractedProfile = profile;
    currentSkills = [...profile.skills];
    renderExtractedProfile(profile);
    document.getElementById('prefs-card').style.display = 'block';
    setStep(3);

  } catch (err) {
    log('✗ Failed: ' + err.message, 'error');
    log('→ You can still add your skills manually below.', 'warn');

    // Open manual entry with blank profile
    const nameHint = uploadedFile.name.replace(/\.[^.]+$/, '').replace(/[-_]/g,'  ')
      .replace(/resume|cv|curriculum|vitae/gi, '').trim();
    extractedProfile = { name: nameHint, title:'', email:'', phone:'', education:[], experience:[], skills:[], experienceYears:0, experienceScore:0.4, educationScore:0.6 };
    currentSkills = [];
    renderExtractedProfile(extractedProfile);
    document.getElementById('prefs-card').style.display = 'block';
    setStep(3);
  }

  btn.disabled = false;
  spin.style.display = 'none';
  lbl.textContent = '🔄 Re-analyze';
}

function showStream(msg, type) {
  const stream = document.getElementById('ai-stream');
  stream.classList.add('show');
  const color = type === 'error' ? 'var(--red-400)' : 'var(--amber-600)';
  stream.innerHTML = `<div style="color:${color}">${msg}</div>`;
}

function renderExtractedProfile(profile) {
  document.getElementById('extracted-profile').classList.add('show');

  const name = (profile.name || '').trim();
  const initials = name.length >= 2
    ? name.split(/\s+/).map(w => w[0]).slice(0, 2).join('').toUpperCase()
    : '?';
  document.getElementById('profile-avatar').textContent = initials;

  const nameEl = document.getElementById('profile-name');
  nameEl.textContent = name || 'Name not detected';
  nameEl.style.color = name ? '' : 'var(--gray-400)';

  const meta = [profile.title, profile.email, profile.phone]
    .filter(Boolean);
  if (profile.experienceYears) meta.push(profile.experienceYears + ' yrs exp');
  if (profile._source) meta.push('(' + profile._source + ')');
  document.getElementById('profile-meta').textContent = meta.join(' · ') || 'Complete your details below';

  renderSkillTagsInProfile();

  document.getElementById('profile-edu').innerHTML =
    (profile.education || []).filter(Boolean).map(e => '<div class="edu-item">' + e + '</div>').join('') ||
    '<div class="edu-item" style="color:var(--gray-400)">Not found in document</div>';

  document.getElementById('profile-exp').innerHTML =
    (profile.experience || []).filter(Boolean).map(e => '<div class="exp-item">' + e + '</div>').join('') ||
    '<div class="exp-item" style="color:var(--gray-400)">Not found in document</div>';
}

function renderSkillTagsInProfile() {
  const el = document.getElementById('profile-skills');
  if (!currentSkills.length) {
    el.innerHTML = '<span style="font-size:12px;color:var(--gray-400)">No skills detected — type yours in the field below</span>';
    return;
  }
  el.innerHTML = currentSkills.map(s =>
    '<span class="skill-tag">' + s +
    '<span class="rm" onclick="removeSkill(\'' + s.replace(/'/g,"\\'") + '\')">\xd7</span></span>'
  ).join('');
}

function removeSkill(s) {
  currentSkills = currentSkills.filter(x => x !== s);
  renderSkillTagsInProfile();
}

function addManualSkill() {
  const inp = document.getElementById('skill-add-input');
  const v = inp.value.trim();
  if (v && !currentSkills.includes(v)) { currentSkills.push(v); renderSkillTagsInProfile(); }
  inp.value = '';
}

function setStep(n) {
  for (let i = 1; i <= 5; i++) {
    const el = document.getElementById('ind-' + i);
    el.classList.remove('active', 'done');
    if (i < n) el.classList.add('done');
    else if (i === n) el.classList.add('active');
  }
}

function switchTab(id) {
  const ids = ['gaps','roadmap','courses','progress','notifs','explain'];
  document.querySelectorAll('.tab').forEach((t, i) => t.classList.toggle('active', ids[i] === id));
  document.querySelectorAll('.tab-panel').forEach(p => p.classList.remove('active'));
  document.getElementById('panel-' + id).classList.add('active');
}

function runPipeline() {
  const btn  = document.getElementById('run-btn');
  const spin = document.getElementById('run-spin');
  const lbl  = document.getElementById('run-label');
  btn.disabled = true;
  spin.style.display = 'block';
  lbl.textContent = 'Generating…';
  setStep(4);

  setTimeout(() => {
    try {
      const role         = document.getElementById('inp-role').value;
      const hoursPerWeek = +document.getElementById('inp-hours').value;
      const deadlineWeeks= +document.getElementById('inp-weeks').value;
      const expScore     = extractedProfile?.experienceScore || 0.4;
      const eduScore     = extractedProfile?.educationScore  || 0.6;
      const name         = extractedProfile?.name || 'You';

      // Gap detection: skill not covered = no current skill is substring or partial match
      const allRequired = ROLE_SKILLS[role] || [];
      const gaps = allRequired.filter(g => {
        const req = g.skill.toLowerCase();
        return !currentSkills.some(s => {
          const us = s.toLowerCase();
          return us === req || us.includes(req) || req.includes(us) || levenSim(us, req) > 0.7;
        });
      }).map(g => {
        // Partial match reduces gap score
        const bestSim = Math.max(0, ...currentSkills.map(s => levenSim(s.toLowerCase(), g.skill.toLowerCase())));
        if (bestSim > 0.4) {
          return { ...g, gap: +(g.gap * (1 - bestSim * 0.5)).toFixed(2), sim: +Math.min(0.8, bestSim).toFixed(2) };
        }
        return g;
      });

      const roadmap = buildRoadmap(gaps, hoursPerWeek, deadlineWeeks, name);

      renderGaps(gaps);
      renderRoadmap(roadmap);
      renderCourses(gaps);
      renderProgress(roadmap, gaps, expScore, eduScore, hoursPerWeek);
      renderNotifications(roadmap, gaps, name);
      renderExplainability(gaps, name);

      const badge = document.getElementById('output-badge');
      badge.textContent = gaps.length + ' gaps found';
      badge.className = 'badge badge-teal';
      setStep(5);
      switchTab('gaps');
    } catch(e) { console.error(e); }

    btn.disabled = false;
    spin.style.display = 'none';
    lbl.textContent = '🔄 Regenerate';
  }, 600);
}

function levenSim(a, b) {
  if (a === b) return 1;
  const longer = a.length > b.length ? a : b;
  const shorter = a.length > b.length ? b : a;
  if (!longer.length) return 1;
  const m = longer.length, n = shorter.length;
  const dp = Array.from({length:m+1},(_,i)=>Array.from({length:n+1},(_,j)=>i===0?j:j===0?i:0));
  for(let i=1;i<=m;i++) for(let j=1;j<=n;j++) dp[i][j]=longer[i-1]===shorter[j-1]?dp[i-1][j-1]:1+Math.min(dp[i-1][j],dp[i][j-1],dp[i-1][j-1]);
  return (longer.length - dp[m][n]) / longer.length;
}

function buildRoadmap(gaps, hoursPerWeek, deadlineWeeks, name) {
  const sorted = [...gaps].sort((a,b) => {
    const pw={high:3,medium:2,low:1};
    return (pw[b.priority]||2)*(1-b.sim) - (pw[a.priority]||2)*(1-a.sim);
  });
  const BUFFER_EVERY = 4;
  const remaining = {};
  sorted.forEach(g => { const c = COURSES[g.skill]; remaining[g.skill] = c ? c.hours : 20; });
  const weeks = [];
  let weekNum = 1;
  while (weekNum <= deadlineWeeks && Object.values(remaining).some(h => h > 0)) {
    const isBuffer = (weekNum % (BUFFER_EVERY + 1) === 0);
    let hoursLeft = isBuffer ? hoursPerWeek * 0.5 : hoursPerWeek;
    const tasks = [], skillsThisWeek = [];
    if (isBuffer) {
      tasks.push({type:'review', title:'Review & consolidation', hours:+(hoursPerWeek*0.4).toFixed(1)});
      tasks.push({type:'project',title:'End-of-module mini project',hours:+(hoursPerWeek*0.6).toFixed(1)});
    } else {
      for (const g of sorted) {
        if ((remaining[g.skill]||0) <= 0 || hoursLeft <= 0) continue;
        const chunk = +Math.min(remaining[g.skill], hoursLeft, hoursPerWeek*0.6).toFixed(1);
        if (chunk < 0.5) continue;
        const c = COURSES[g.skill] || {title:g.skill+' Course',platform:'Udemy',rating:4.5};
        tasks.push({type:'course',title:c.title.slice(0,46),skill:g.skill,hours:chunk,course:c});
        remaining[g.skill] = +(remaining[g.skill]-chunk).toFixed(2);
        hoursLeft = +(hoursLeft-chunk).toFixed(2);
        skillsThisWeek.push(g.skill);
        if (remaining[g.skill] <= 0) {
          const projH = +Math.min(hoursLeft,2).toFixed(1);
          if (projH >= 0.5) { tasks.push({type:'project',title:'Mini project: Apply '+g.skill,hours:projH,skill:g.skill}); hoursLeft=+(hoursLeft-projH).toFixed(2); }
        }
      }
    }
    const coveredCount = Object.values(remaining).filter(h=>h<=0).length;
    const pct = sorted.length ? coveredCount/sorted.length : 0;
    let milestone = null;
    if (Math.abs(pct-.25)<.12 && coveredCount>0) milestone='🏅 25% complete';
    else if (Math.abs(pct-.5)<.12) milestone='🏆 Halfway there!';
    else if (Math.abs(pct-.75)<.12) milestone='🚀 75% complete';
    else if (pct>=.99) milestone='🎓 All '+sorted.length+' skills done!';
    weeks.push({weekNum,isBuffer,tasks,milestone,theme:skillsThisWeek.slice(0,2).join(' + ')||'Review',totalHours:+tasks.reduce((s,t)=>s+t.hours,0).toFixed(1)});
    weekNum++;
  }
  return {name, weeks, totalWeeks:weeks.length, hoursPerWeek, deadlineWeeks, gaps:sorted};
}

function renderRoadmap(roadmap) {
  const {weeks} = roadmap;
  const totalH = weeks.reduce((s,w)=>s+w.totalHours,0).toFixed(0);
  const weeksHtml = weeks.map(w => {
    const tasks = w.tasks.slice(0,3).map(t=>{
      const cls=t.type==='course'?'course':t.type==='project'?'project':'review';
      const ico=t.type==='course'?'📚':t.type==='project'?'🔨':'🔁';
      return `<div class="week-task ${cls}">${ico} ${(t.title||'').slice(0,30)} <b>${t.hours}h</b></div>`;
    }).join('');
    const ms = w.milestone?`<div class="week-task milestone">🏆 ${w.milestone}</div>`:'';
    return `<div class="week-col"><div class="week-header${w.isBuffer?' buffer':''}">W${w.weekNum}${w.isBuffer?' ⏸':''}</div>${tasks}${ms}</div>`;
  }).join('');
  document.getElementById('roadmap-output').innerHTML = `
    <div style="display:flex;gap:8px;margin-bottom:14px;flex-wrap:wrap">
      <span class="tag" style="background:var(--blue-50);color:var(--blue-600)">${weeks.length} weeks</span>
      <span class="tag" style="background:var(--teal-50);color:var(--teal-600)">${totalH}h total</span>
      <span class="tag" style="background:var(--amber-50);color:var(--amber-600)">${weeks.filter(w=>w.isBuffer).length} review weeks</span>
      <span class="tag" style="background:var(--green-50);color:var(--green-600)">${weeks.reduce((s,w)=>s+w.tasks.filter(t=>t.type==='project').length,0)} mini-projects</span>
    </div>
    <div class="roadmap-scroll"><div class="roadmap-timeline">${weeksHtml}</div></div>
    <div style="display:flex;gap:10px;margin-top:10px;font-size:10px;flex-wrap:wrap">
      <span><span style="display:inline-block;width:9px;height:9px;background:var(--blue-100);border-radius:2px;margin-right:3px"></span>Course</span>
      <span><span style="display:inline-block;width:9px;height:9px;background:var(--green-50);border-radius:2px;margin-right:3px"></span>Mini project</span>
